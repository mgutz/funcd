mixinTag = (tag) ->
  Funcd::[tag] = (attributes, inner) ->
    options = tag: tag, parseBody: tag != 'textarea', parseAttributes: true
    @_outerHtml options, attributes, inner

mixinShortTag = (tag) ->
  Funcd::[tag] = (attributes) ->

    if _.isObject(attributes)
